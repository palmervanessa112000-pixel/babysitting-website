<script>
let bookedDates = []; // add dates like "2026-04-10"

function checkAvailability() {
    let date = document.getElementById("date").value;
    let status = document.getElementById("status");

    if (!date) {
        status.innerHTML = "Please select a date";
        status.style.color = "black";
        return;
    }

    if (bookedDates.includes(date)) {
        status.innerHTML = "🔴 Not Available";
        status.style.color = "red";
    } else {
        status.innerHTML = "🟢 Available";
        status.style.color = "green";
    }
}

function sendWhatsApp() {
    let date = document.getElementById("date").value;
    let time = document.getElementById("time").value;

    if (!date) {
        alert("Please select a date first");
        return;
    }

    let message = `Hi, I want to book babysitting for ${date} at ${time}`;
    window.open(`https://wa.me/18768032970?text=${message}`, "_blank");
}
